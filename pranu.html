# Import necessary libraries
import numpy as np
import matplotlib.pyplot as plt

# Define the x values from e to 10
x = np.linspace(np.e, 10, 100)
# Define the y values as the square of x
y = x**2

# Create the plot
plt.figure(figsize=(8, 5))
plt.plot(x, y, label="y = x^2", color="blue")

# Customize the plot
plt.title("Line Plot of y = x^2 from e to 10", fontsize=14)
plt.xlabel("x values", fontsize=12)
plt.ylabel("y values (x^2)", fontsize=12)
plt.grid(True, linestyle="--", alpha=0.7)
plt.legend(fontsize=10)
plt.tight_layout()

# Display the plot
plt.show()

# Import necessary libraries
import numpy as np
import matplotlib.pyplot as plt

# Data for the subplots
x = np.linspace(0, 10, 100)
y = x**2

categories = ['A', 'B', 'C', 'D']
values = [5, 7, 3, 9]
scatter_x = [1, 2, 3, 4, 5]
scatter_y = [2, 4, 6, 8, 10]
pie_labels = ['Python', 'Java', 'C++', 'JavaScript']
pie_sizes = [40, 25, 20, 15]
pie_colors = ['gold', 'skyblue', 'lightgreen', 'orange']

# Create a 2x2 subplot layout
fig, axs = plt.subplots(2, 2, figsize=(12, 10))

# Line chart in the first subplot
axs[0, 0].plot(x, y, color="blue", label="y = x^2")
axs[0, 0].set_title("Line Chart", fontsize=12)
axs[0, 0].set_xlabel("x values")
axs[0, 0].set_ylabel("y values")
axs[0, 0].legend()
axs[0, 0].grid(True)

# Bar chart in the second subplot
axs[0, 1].bar(categories, values, color=['red', 'green', 'blue', 'orange'])
axs[0, 1].set_title("Bar Chart", fontsize=12)
axs[0, 1].set_xlabel("Categories")
axs[0, 1].set_ylabel("Values")
axs[0, 1].grid(axis='y', linestyle="--", alpha=0.7)

# Scatter plot in the third subplot
axs[1, 0].scatter(scatter_x, scatter_y, color="purple")
axs[1, 0].set_title("Scatter Plot", fontsize=12)
axs[1, 0].set_xlabel("x values")
axs[1, 0].set_ylabel("y values")
for xi, yi in zip(scatter_x, scatter_y):
    axs[1, 0].text(xi, yi, f"({xi}, {yi})", fontsize=9, ha='right', va='bottom')
axs[1, 0].grid(True)

# Pie chart in the fourth subplot
axs[1, 1].pie(pie_sizes, labels=pie_labels, colors=pie_colors, autopct='%1.1f%%', startangle=90, wedgeprops={'edgecolor': 'black'})
axs[1, 1].set_title("Pie Chart", fontsize=12)

# Adjust layout
plt.tight_layout()
plt.show()

# Importing the necessary library
import matplotlib.pyplot as plt

# Data for the histogram
data = [1, 2, 2, 3, 3, 3, 4, 4, 4, 4, 5]

# Creating the histogram
plt.figure(figsize=(8, 5))
plt.hist(data, bins=range(1, 7), color='skyblue', edgecolor='black', align='left', rwidth=0.8)

# Customizing the histogram
plt.title('Frequency Distribution of Data', fontsize=14)
plt.xlabel('Values', fontsize=12)
plt.ylabel('Frequency', fontsize=12)
plt.xticks(range(1, 6))

# Display the histogram
plt.show()
